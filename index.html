#include // กำหนดขาพินสำหรับ HC-SR04 const int trigPin = 13; // D7 const int echoPin = 15; // D8 // กำหนดขาพินสำหรับคุยกับ SIM800L (RX, TX) SoftwareSerial sim800l(0, 2); // D3 เป็น RX, D4 เป็น TX // ตัวแปรสำหรับเซ็นเซอร์อัลตร้าโซนิค long duration; int distance; int thresholdDistance = 100; // ระยะตรวจจับ 100 เซนติเมตร // เบอร์โทรศัพท์ที่ต้องการให้โทรไปแจ้งเตือน (เปลี่ยนเป็นเบอร์ของคุณครู) String phoneNumber = "+66891234567"; // ตัวแปรป้องกันการโทรซ้ำรัวๆ unsigned long lastCallTime = 0; const unsigned long callCooldown = 60000; // หน่วงเวลา 60 วินาทีถึงจะโทรซ้ำได้ void setup() { Serial.begin(115200); sim800l.begin(9600); // ความเร็วมาตรฐานของ SIM800L pinMode(trigPin, OUTPUT); pinMode(echoPin, INPUT); Serial.println("System Initializing..."); delay(5000); // รอให้โมดูล SIM จับสัญญาณเครือข่ายให้เสร็จ Serial.println("Smart Security System Ready!"); } void loop() { // 1. กระบวนการยิงคลื่นอัลตร้าโซนิคเพื่อวัดระยะ digitalWrite(trigPin, LOW); delayMicroseconds(2); digitalWrite(trigPin, HIGH); delayMicroseconds(10); digitalWrite(trigPin, LOW); // 2. อ่านเวลาและคำนวณเป็นเซนติเมตร duration = pulseIn(echoPin, HIGH); distance = duration * 0.034 / 2; Serial.print("Distance: "); Serial.print(distance); Serial.println(" cm"); // 3. ตรวจสอบผู้บุกรุก และตรวจสอบว่าพ้นระยะหน่วงเวลา(Cooldown)หรือยัง if (distance > 0 && distance < thresholdDistance) { if (millis() - lastCallTime > callCooldown || lastCallTime == 0) { Serial.println("🚨 INTRUDER DETECTED! Calling Owner..."); makeCall(); lastCallTime = millis(); // บันทึกเวลาที่โทรออกล่าสุด } else { Serial.println("Intruder detected, but call is on cooldown."); } } delay(1000); // หน่วงเวลาอ่านค่ารอบต่อไป 1 วินาที } // ฟังก์ชันสั่ง SIM800L โทรออก void makeCall() { sim800l.println("AT"); // เช็คการเชื่อมต่อ delay(1000); sim800l.print("ATD"); // คำสั่งโทรออก (ATD) sim800l.print(phoneNumber); sim800l.println(";"); // ต้องปิดท้ายด้วยเครื่องหมายอัฒภาค (;) สำหรับการโทรเสียง delay(20000); // ปล่อยให้สายดัง 20 วินาที sim800l.println("ATH"); // คำสั่งวางสาย (ATH) Serial.println("Call Ended."); }
